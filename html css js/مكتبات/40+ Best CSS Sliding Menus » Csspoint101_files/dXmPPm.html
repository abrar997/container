function menuEffects() {

  function hasParentClass(e, classname) {
    if (e === document) {
      return false;
    } else if(e.classList.contains(classname)) {
      return true;
    } else {
      return e.parentNode && hasParentClass(e.parentNode, classname);
    }
  }


  let container = document.getElementById('sdcnt'),
      buttons = document.querySelectorAll('#sd-trig > button');

  function bodyClickFn(evt) {
    if (!hasParentClass(evt.target, 'sdmenu')) {
      container.classList.remove('sdmenu-open');
      document.removeEventListener('click', bodyClickFn);
    }
  }

  buttons.forEach(function (ele) {
    let effect = ele.getAttribute('effect');
    ele.addEventListener('click', function(evt){
      evt.stopPropagation();
      evt.preventDefault();
      container.className = 'sdcnt';
      container.classList.add(effect);
      setTimeout(function () {
        container.classList.add('sdmenu-open');
      }, 25);
      document.addEventListener('click', bodyClickFn);
    });
  });

}
